1. Symbol table:

#include <stdio.h>
#include <string.h>

struct table {
    char var[10];
    int value;
} tbl[20];

int n = 0;

// Search function
int search(char v[]) {
    for(int i = 0; i < n; i++) {
        if(strcmp(tbl[i].var, v) == 0)
            return i;
    }
    return -1;
}

// Insert function
void insert() {
    if(n >= 20) {
        printf("Table Full\n");
        return;
    }

    printf("Enter variable and value: ");
    scanf("%s %d", tbl[n].var, &tbl[n].value);

    if(search(tbl[n].var) != -1) {
        printf("Variable already exists\n");
        return;
    }

    n++;
}

// Modify function
void modify() {
    char v[10];
    printf("Enter variable to modify: ");
    scanf("%s", v);

    int pos = search(v);
    if(pos == -1) {
        printf("Not found\n");
        return;
    }

    printf("Enter new value: ");
    scanf("%d", &tbl[pos].value);
}

// Display function
void display() {
    printf("\nVar\tValue\n");
    for(int i = 0; i < n; i++)
        printf("%s\t%d\n", tbl[i].var, tbl[i].value);
}

int main() {
    int ch;

    while(1) {
        printf("\n1.Insert 2.Modify 3.Search 4.Display 5.Exit\n");
        scanf("%d", &ch);

        if(ch == 1) insert();
        else if(ch == 2) modify();
        else if(ch == 3) {
            char v[10];
            printf("Enter variable: ");
            scanf("%s", v);

            int pos = search(v);
            if(pos == -1)
                printf("Not found\n");
            else
                printf("Value = %d\n", tbl[pos].value);
        }
        else if(ch == 4) display();
        else break;
    }

    return 0;
}


2. Lexical analyzer to identify a few patterns

#include <stdio.h>
#include <ctype.h>
#include <string.h>

int isKeyword(char str[]) {
    char *kw[] = {"int","float","char","for","while","if","else","return"};
    for(int i = 0; i < 8; i++) {
        if(strcmp(str, kw[i]) == 0)
            return 1;
    }
    return 0;
}

int main() {
    char str[50];
    int i = 0;
    char ch;

    printf("Enter code (Ctrl+D to end):\n");

    while((ch = getchar()) != EOF) {

        // IDENTIFIERS / KEYWORDS
        if(isalpha(ch)) {
            str[i++] = ch;
            while(isalnum(ch = getchar()))
                str[i++] = ch;

            str[i] = '\0';
            i = 0;

            if(isKeyword(str))
                printf("%s -> Keyword\n", str);
            else
                printf("%s -> Identifier\n", str);
        }

        // NUMBERS (CONSTANTS)
        else if(isdigit(ch)) {
            int num = ch - '0';
            while(isdigit(ch = getchar()))
                num = num * 10 + (ch - '0');

            printf("%d -> Constant\n", num);
        }

        // OPERATORS / SPECIAL CHARACTERS
        else if(ch=='+'||ch=='-'||ch=='*'||ch=='/'||ch=='='||ch=='<'||ch=='>') {
            printf("%c -> Operator\n", ch);
        }

        // IGNORE SPACES
    }

    return 0;
}


3. First functions


#include <stdio.h>
#include <ctype.h>
#include <string.h>

char prod[10][10];
char first[10];
int n = 0, count;

void add(char c) {
    for(int i=0;i<n;i++)
        if(first[i]==c) return;
    first[n++] = c;
}

void FIRST(char c) {
    if(!isupper(c)) {
        add(c);
        return;
    }

    for(int i=0;i<count;i++) {
        if(prod[i][0] == c) {
            char next = prod[i][2];

            if(!isupper(next))
                add(next);
            else
                FIRST(next);
        }
    }
}

int main() {
    printf("Enter number of productions: ");
    scanf("%d", &count);

    printf("Enter productions (E=E+T format):\n");
    for(int i=0;i<count;i++)
        scanf("%s", prod[i]);

    char ch;
    printf("Enter non-terminal: ");
    scanf(" %c", &ch);

    FIRST(ch);

    printf("FIRST(%c) = { ", ch);
    for(int i=0;i<n;i++)
        printf("%c ", first[i]);
    printf("}\n");

    return 0;
}



4. Recursive descent parser
S → ( L ) | a  
L → L , S | S


#include <stdio.h>
#include <stdlib.h>
#include <string.h>

const char *input;
int pos = 0;

// Function prototypes
int S(), L(), Ldash();

void error() {
    printf("Error at position %d\n", pos);
    exit(1);
}

char lookahead() {
    return input[pos];
}

// S → (L) | a
int S() {
    if (lookahead() == 'a') {
        pos++;
        return 1;
    }
    else if (lookahead() == '(') {
        pos++;
        if (!L()) error();
        if (lookahead() == ')') {
            pos++;
            return 1;
        } else {
            error();
        }
    }
    return 0;
}

// L → S L'
int L() {
    if (!S()) return 0;
    return Ldash();
}

// L' → , S L' | ε
int Ldash() {
    if (lookahead() == ',') {
        pos++;
        if (!S()) error();
        return Ldash();
    }
    return 1; // epsilon
}

int main() {
    char buffer[100];

    printf("Enter input string: ");
    scanf("%s", buffer);

    input = buffer;
    pos = 0;

    if (S() && input[pos] == '\0')
        printf("String accepted\n");
    else
        error();

    return 0;
}



5. Recursive Descent parser
E → E + T | T  
T → T * F | F  
F → ( E ) | id


#include <stdio.h>
#include <stdlib.h>
#include <string.h>

char input[100];
int pos = 0;

char lookahead() {
    return input[pos];
}

void error() {
    printf("Error at position %d\n", pos);
    exit(1);
}

// Function declarations
int E(), Edash(), T(), Tdash(), F();

// E → T E'
int E() {
    if (T())
        return Edash();
    return 0;
}

// E' → + T E' | ε
int Edash() {
    if (lookahead() == '+') {
        pos++;
        if (T())
            return Edash();
        else
            error();
    }
    return 1; // epsilon
}

// T → F T'
int T() {
    if (F())
        return Tdash();
    return 0;
}

// T' → * F T' | ε
int Tdash() {
    if (lookahead() == '*') {
        pos++;
        if (F())
            return Tdash();
        else
            error();
    }
    return 1; // epsilon
}

// F → (E) | i
int F() {
    if (lookahead() == 'i') {
        pos++;
        return 1;
    }
    else if (lookahead() == '(') {
        pos++;
        if (E()) {
            if (lookahead() == ')') {
                pos++;
                return 1;
            } else {
                error();
            }
        }
    }
    return 0;
}

int main() {
    printf("Enter input string: ");
    scanf("%s", input);

    pos = 0;

    if (E() && input[pos] == '\0')
        printf("String Accepted\n");
    else
        error();

    return 0;
}

#include <stdlib.h>
#define YYSTYPE double

int yylex();
int yyerror();
%}

%token NUM
%left '+' '-'
%left '*' '/'
%right UMINUS

%%
Statement:
      E '\n'        { printf("Answer: %g\n", $1); }
    | Statement E '\n'
    ;

E:
      E '+' E       { $$ = $1 + $3; }
    | E '-' E       { $$ = $1 - $3; }
    | E '*' E       { $$ = $1 * $3; }
    | E '/' E       { $$ = $1 / $3; }
    | '-' E %prec UMINUS { $$ = -$2; }
    | NUM
    ;
%%

int main() {
    printf("Enter expression:\n");
    yyparse();
    return 0;
}

int yyerror() {
    printf("Invalid Expression\n");
    return 0;
}

7. Implementation of STACK-------

#include <stdio.h>
#include <stdlib.h>

#define SIZE 5

struct stack {
    int s[SIZE];
    int top;
} st;

// Check if stack is full
int isFull() {
    return (st.top == SIZE - 1);
}

// Check if stack is empty
int isEmpty() {
    return (st.top == -1);
}

// Push operation
void push(int item) {
    if (isFull()) {
        printf("Stack Overflow\n");
        return;
    }
    st.s[++st.top] = item;
}

// Pop operation
int pop() {
    if (isEmpty()) {
        printf("Stack Underflow\n");
        return -1;
    }
    return st.s[st.top--];
}

// Display stack
void display() {
    if (isEmpty()) {
        printf("Stack is Empty\n");
        return;
    }
    printf("Stack elements:\n");
    for (int i = st.top; i >= 0; i--)
        printf("%d\n", st.s[i]);
}

int main() {
    int choice, item;

    st.top = -1;

    while (1) {
        printf("\n1.Push 2.Pop 3.Display 4.Exit\n");
        scanf("%d", &choice);

        switch (choice) {
            case 1:
                printf("Enter element: ");
                scanf("%d", &item);
                push(item);
                break;

            case 2:
                item = pop();
                if (item != -1)
                    printf("Popped: %d\n", item);
                break;

            case 3:
                display();
                break;

            case 4:
                exit(0);

            default:
                printf("Invalid choice\n");
        }
    }
}
